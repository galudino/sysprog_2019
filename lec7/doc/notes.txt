struct typetable {
    size_t width;
    void *(*copy)(const void *);
    void (*dtor)(void *);
    void (*swap)(void *, void *);
    int (*compare)(const void *, const void *);
    void (*print)(const void *, FILE *);
};

typedef char *str;
void *str_copy(const void *arg);
void str_dtor(void *arg);
void str_swap(void *c1, void *c2);
